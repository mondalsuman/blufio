**lib.rs:** Define `BridgeManager` struct (lifecycle manager -- routing implemented in Task 2):
```rust
pub struct BridgeManager {
    groups: HashMap<String, BridgeGroupConfig>,
}

impl BridgeManager {
    pub fn new(groups: HashMap<String, BridgeGroupConfig>) -> Self {
        Self { groups }
    }

    pub fn is_empty(&self) -> bool {
        self.groups.is_empty()
    }

    /// Check if a message should be bridged (not already bridged, passes filters).
    pub fn should_bridge(&self, channel: &str, sender_id: &str, is_bridged: bool, is_bot: bool) -> Vec<(String, Vec<String>)> {
        if is_bridged { return vec![]; }
        let mut targets = vec![];
        for (group_name, group) in &self.groups {
            if !group.channels.contains(&channel.to_string()) { continue; }
            if group.exclude_bots && is_bot { continue; }
            if !group.include_users.is_empty() && !group.include_users.contains(&sender_id.to_string()) { continue; }
            let target_channels: Vec<String> = group.channels.iter()
                .filter(|c| c.as_str() != channel)
                .cloned()
                .collect();
            if !target_channels.is_empty() {
                targets.push((group_name.clone(), target_channels));
            }
        }
        targets
    }
}
```

**router.rs:** Implement the bridge event loop:
```rust
use crate::formatter;
use blufio_bus::events::{BusEvent, ChannelEvent};
use blufio_bus::EventBus;
use blufio_config::model::BridgeGroupConfig;
use blufio_core::types::OutboundMessage;
use std::collections::HashMap;
use std::sync::Arc;
use tokio::sync::mpsc;
use tracing::{debug, info, warn};

/// Outbound bridge message ready to be sent to a target channel.
pub struct BridgedMessage {
    pub target_channel: String,
    pub content: String,
}

/// Run the bridge event loop.
///
/// Subscribes to the event bus, filters messages by bridge group membership,
/// formats with sender attribution, and sends to the bridge_tx channel.
/// The caller (serve.rs) reads from bridge_rx and dispatches via ChannelMultiplexer.
pub async fn run_bridge_loop(
    bus: Arc<EventBus>,
    groups: HashMap<String, BridgeGroupConfig>,
    bridge_tx: mpsc::Sender<BridgedMessage>,
) {
    let manager = super::BridgeManager::new(groups);
    if manager.is_empty() {
        info!("no bridge groups configured, bridge loop not started");
        return;
    }

    let mut rx = bus.subscribe();
    info!("bridge loop started, listening for channel events");

    loop {
        match rx.recv().await {
            Ok(BusEvent::Channel(ChannelEvent::MessageReceived {
                channel,
                sender_id,
                content: Some(content),
                sender_name,
                is_bridged,
                ..
            })) => {
                // Determine target channels for this message.
                let targets = manager.should_bridge(&channel, &sender_id, is_bridged, false);

                if targets.is_empty() {
                    continue;
                }

                let display_name = sender_name.as_deref().unwrap_or(&sender_id);
                let formatted = formatter::format_bridged_message(&channel, display_name, &content);

                for (_group_name, target_channels) in targets {
                    for target in target_channels {
                        debug!(
                            source = %channel,
                            target = %target,
                            sender = %display_name,
                            "bridging message"
                        );
                        if bridge_tx.send(BridgedMessage {
                            target_channel: target,
                            content: formatted.clone(),
                        }).await.is_err() {
                            warn!("bridge outbound channel closed");
                            return;
                        }
                    }
                }
            }
            Ok(_) => {} // Ignore non-channel events and events without content.
            Err(tokio::sync::broadcast::error::RecvError::Lagged(n)) => {
                warn!(missed = n, "bridge subscriber lagged behind event bus");
            }
            Err(tokio::sync::broadcast::error::RecvError::Closed) => {
                info!("event bus closed, bridge loop stopping");
                return;
            }
        }
    }
}
```

```rust
#[cfg(feature = "bridge")]
use blufio_bridge::router::BridgedMessage;

// After all channels are added to mux and mux.connect_all() is called:
#[cfg(feature = "bridge")]
{
    if !config.bridge.is_empty() {
        let (mut bridge_rx, _bridge_handle) = blufio_bridge::spawn_bridge(
            bus.clone(),  // Arc<EventBus> from earlier in the function
            config.bridge.clone(),
        );
        info!(groups = config.bridge.len(), "cross-channel bridge started");

        // Spawn a task that reads bridged messages and dispatches via ChannelMultiplexer.
        let mux_clone = mux.clone(); // or however the mux is shared (Arc)
        tokio::spawn(async move {
            while let Some(bridged) = bridge_rx.recv().await {
                let msg = OutboundMessage {
                    session_id: None,
                    channel: bridged.target_channel.clone(),
                    content: bridged.content,
                    reply_to: None,
                    parse_mode: None,
                    metadata: Some(serde_json::json!({
                        "is_bridged": true
                    }).to_string()),
                };
                if let Err(e) = mux_clone.send(&bridged.target_channel, msg).await {
                    warn!(
                        target = %bridged.target_channel,
                        error = %e,
                        "failed to deliver bridged message"
                    );
                }
            }
        });
    } else {
        info!("cross-channel bridge skipped (no bridge groups configured)");
    }
}
```

IMPORTANT: The outbound bridged messages must include `is_bridged: true` in metadata. When adapters publish `ChannelEvent::MessageReceived` for these messages, they should check metadata for `is_bridged` and set the event field accordingly. This prevents the bridge from re-bridging its own forwarded messages (infinite loop prevention).

The exact integration depends on how the ChannelMultiplexer and adapters interact with the event bus. The executor should ensure that:
1. When an adapter receives a message from the network, it publishes ChannelEvent::MessageReceived with is_bridged=false.
2. When the bridge sends a message to an adapter, the adapter does NOT publish a new ChannelEvent::MessageReceived for it. OR if it does, the metadata `is_bridged` flag is checked and set to true in the event.
3. The bridge subscriber skips events where is_bridged=true.
