Create `contrib/blufio@.service` based on the existing `contrib/blufio.service` with the following template adaptations:

```ini
# Blufio AI Agent - multi-instance systemd template
#
# Usage:
#   1. Run contrib/blufio-instance-setup.sh <instance-name>
#   2. Edit /etc/blufio/instances/<instance-name>/config.toml
#   3. systemctl enable blufio@<instance-name>
#   4. systemctl start blufio@<instance-name>
#
# Examples:
#   systemctl start blufio@personal
#   systemctl start blufio@work
#   systemctl status blufio@personal
#   journalctl -u blufio@personal -f

[Unit]
Description=Blufio AI Agent (%i)
Documentation=https://github.com/blufio/blufio
After=network-online.target
Wants=network-online.target

[Service]
Type=notify
NotifyAccess=main
ExecStartPre=-/var/lib/blufio/instances/%i/hooks/pre-start.sh
ExecStart=/usr/local/bin/blufio serve
ExecStopPost=-/var/lib/blufio/instances/%i/hooks/post-stop.sh
Restart=on-failure
RestartSec=5s
StartLimitBurst=3
StartLimitIntervalSec=60
TimeoutStartSec=90
TimeoutStopSec=30
WatchdogSec=30

User=blufio
Group=blufio
WorkingDirectory=/var/lib/blufio/instances/%i

Environment=RUST_LOG=blufio=info
Environment=BLUFIO_CONFIG=/etc/blufio/instances/%i/config.toml
EnvironmentFile=-/etc/blufio/instances/%i/environment

# Security hardening (same as single-instance service)
NoNewPrivileges=yes
PrivateTmp=yes
PrivateDevices=yes
ProtectSystem=strict
ProtectHome=yes
ReadWritePaths=/var/lib/blufio/instances/%i
ProtectKernelTunables=yes
ProtectKernelModules=yes
RestrictAddressFamilies=AF_UNIX AF_INET AF_INET6
LimitNOFILE=65536

# Memory limit per instance
MemoryMax=256M
MemoryHigh=200M

[Install]
WantedBy=multi-user.target
```

Key differences from contrib/blufio.service:
- Description includes `(%i)` for instance identification in `systemctl list-units`
- WorkingDirectory uses `/var/lib/blufio/instances/%i` (not `/var/lib/blufio`)
- BLUFIO_CONFIG points to instance-specific config: `/etc/blufio/instances/%i/config.toml`
- EnvironmentFile reads from `/etc/blufio/instances/%i/environment`
- ReadWritePaths scoped to instance directory only
- Hook paths are instance-specific: `/var/lib/blufio/instances/%i/hooks/`

Create `contrib/blufio-instance-setup.sh` — a helper script that creates the directory structure and default config for a new Blufio instance.

```bash
#!/usr/bin/env bash
# blufio-instance-setup.sh — Create a new Blufio instance for multi-instance systemd deployment.
#
# Usage:
#   sudo ./contrib/blufio-instance-setup.sh <instance-name> [--port PORT]
#
# Creates:
#   /etc/blufio/instances/<name>/config.toml    — Instance configuration
#   /etc/blufio/instances/<name>/environment     — Environment variables (secrets)
#   /var/lib/blufio/instances/<name>/            — Instance working directory
#   /var/lib/blufio/instances/<name>/plugins/    — WASM plugins
#   /var/lib/blufio/instances/<name>/hooks/      — Optional lifecycle hooks
#
# After setup:
#   1. Edit /etc/blufio/instances/<name>/config.toml (set adapters, providers)
#   2. Edit /etc/blufio/instances/<name>/environment (set API keys)
#   3. systemctl enable --now blufio@<name>

set -euo pipefail

INSTANCE_NAME="${1:-}"
PORT="${3:-}"  # --port PORT

# Parse --port flag
shift || true
while [[ $# -gt 0 ]]; do
    case "$1" in
        --port)
            PORT="$2"
            shift 2
            ;;
        *)
            echo "Unknown option: $1" >&2
            exit 1
            ;;
    esac
done

if [[ -z "$INSTANCE_NAME" ]]; then
    echo "Usage: $0 <instance-name> [--port PORT]"
    echo ""
    echo "Examples:"
    echo "  $0 personal"
    echo "  $0 work --port 3001"
    echo "  $0 team --port 3002"
    exit 1
fi

# Validate instance name (alphanumeric, hyphens, underscores).
if ! [[ "$INSTANCE_NAME" =~ ^[a-zA-Z0-9_-]+$ ]]; then
    echo "Error: instance name must be alphanumeric (with hyphens/underscores)" >&2
    exit 1
fi

CONFIG_DIR="/etc/blufio/instances/${INSTANCE_NAME}"
DATA_DIR="/var/lib/blufio/instances/${INSTANCE_NAME}"
DEFAULT_PORT="${PORT:-3000}"

# Check if instance already exists.
if [[ -d "$CONFIG_DIR" ]] || [[ -d "$DATA_DIR" ]]; then
    echo "Error: instance '${INSTANCE_NAME}' already exists" >&2
    echo "  Config: ${CONFIG_DIR}"
    echo "  Data:   ${DATA_DIR}"
    exit 1
fi

echo "Creating Blufio instance: ${INSTANCE_NAME}"
echo "  Config: ${CONFIG_DIR}"
echo "  Data:   ${DATA_DIR}"
echo "  Port:   ${DEFAULT_PORT}"
echo ""

# Create directories.
mkdir -p "${CONFIG_DIR}"
mkdir -p "${DATA_DIR}/plugins"
mkdir -p "${DATA_DIR}/hooks"

# Create default config.toml with instance-specific port.
cat > "${CONFIG_DIR}/config.toml" << TOML
# Blufio instance configuration: ${INSTANCE_NAME}
# See deploy/.env.example for environment variable reference.

[agent]
name = "blufio-${INSTANCE_NAME}"

[gateway]
enabled = true
host = "127.0.0.1"
port = ${DEFAULT_PORT}

[storage]
database_path = "${DATA_DIR}/blufio.db"

[plugin]
skill_dir = "${DATA_DIR}/plugins"
TOML

# Create environment file template.
cat > "${CONFIG_DIR}/environment" << ENV
# Environment variables for blufio@${INSTANCE_NAME}
# Loaded by systemd EnvironmentFile= directive.
#
# Uncomment and set your API keys:
# ANTHROPIC_API_KEY=sk-ant-...
# OPENAI_API_KEY=sk-...
# TELEGRAM_BOT_TOKEN=...
# DISCORD_TOKEN=...
RUST_LOG=blufio=info
ENV

# Set ownership (create blufio user if needed).
if id blufio &>/dev/null; then
    chown -R blufio:blufio "${DATA_DIR}"
    chown -R root:blufio "${CONFIG_DIR}"
    chmod 750 "${CONFIG_DIR}"
    chmod 640 "${CONFIG_DIR}/environment"
else
    echo "Warning: 'blufio' user does not exist. Run:"
    echo "  useradd --system --shell /usr/sbin/nologin --home-dir /var/lib/blufio blufio"
fi

echo ""
echo "Instance '${INSTANCE_NAME}' created successfully."
echo ""
echo "Next steps:"
echo "  1. Edit ${CONFIG_DIR}/config.toml"
echo "  2. Edit ${CONFIG_DIR}/environment (add API keys)"
echo "  3. systemctl enable --now blufio@${INSTANCE_NAME}"
echo ""
echo "Management:"
echo "  systemctl status blufio@${INSTANCE_NAME}"
echo "  journalctl -u blufio@${INSTANCE_NAME} -f"
```

Make the script executable: `chmod +x contrib/blufio-instance-setup.sh`

Key design decisions:
- Instance name validated to be alphanumeric with hyphens/underscores (safe for systemd %i)
- Config directory under /etc/blufio/instances/{name}/ (config is separate from data)
- Data directory under /var/lib/blufio/instances/{name}/ (database, plugins, hooks)
- Environment file permissions restricted (640, root:blufio) since it contains secrets
- Default port 3000, overridable via --port flag for multi-instance setups
- Creates blufio user ownership if user exists, warns otherwise
- Refuses to overwrite existing instances
