       ```rust
       use ring::digest;

       /// Compute a SHA-256 hash pin for a tool definition.
       ///
       /// The hash is computed over a canonical JSON representation of the tool's
       /// name, description, and input schema. JSON keys are sorted to ensure
       /// deterministic hashing regardless of serialization order.
       pub fn compute_tool_pin(
           name: &str,
           description: Option<&str>,
           schema: &serde_json::Value,
       ) -> String {
           let canonical = serde_json::json!({
               "description": description.unwrap_or(""),
               "input_schema": canonicalize_json(schema),
               "name": name,
           });
           let bytes = serde_json::to_vec(&canonical).expect("canonical JSON serialization");
           let digest = digest::digest(&digest::SHA256, &bytes);
           hex::encode(digest.as_ref())
       }

       /// Recursively sort JSON object keys for canonical representation.
       fn canonicalize_json(value: &serde_json::Value) -> serde_json::Value {
           match value {
               serde_json::Value::Object(map) => {
                   let sorted: serde_json::Map<String, serde_json::Value> = map
                       .iter()
                       .map(|(k, v)| (k.clone(), canonicalize_json(v)))
                       .collect::<std::collections::BTreeMap<_, _>>()
                       .into_iter()
                       .collect();
                   serde_json::Value::Object(sorted)
               }
               serde_json::Value::Array(arr) => {
                   serde_json::Value::Array(arr.iter().map(canonicalize_json).collect())
               }
               _ => value.clone(),
           }
       }

       /// Verify a tool's current definition against a stored pin.
       /// Returns true if the pin matches, false if schema has mutated.
       pub fn verify_pin(
           name: &str,
           description: Option<&str>,
           schema: &serde_json::Value,
           stored_pin: &str,
       ) -> bool {
           compute_tool_pin(name, description, schema) == stored_pin
       }
       ```

    3. **sanitize.rs**: Create `crates/blufio-mcp-client/src/sanitize.rs`:

       ```rust
       use regex::Regex;
       use std::sync::LazyLock;

       /// Maximum length for sanitized external tool descriptions.
       const MAX_DESCRIPTION_LEN: usize = 200;

       /// Regex matching instruction-like patterns in tool descriptions.
       /// These are stripped to prevent prompt injection via external tool descriptions.
       static INSTRUCTION_PATTERN: LazyLock<Regex> = LazyLock::new(|| {
           Regex::new(
               r"(?i)(?:you must|you should|you are|always|never|important:|note:|remember:|warning:|caution:)[^.!?\n]*[.!?\n]?"
           ).expect("valid instruction pattern regex")
       });

       /// Sanitize an external tool description for safe inclusion in prompts.
       ///
       /// 1. Strips instruction-like patterns that could influence LLM behavior
       /// 2. Trims whitespace and collapses multiple spaces
       /// 3. Caps at 200 characters with "..." suffix
       /// 4. Prefixes with "[External: {server_name}]"
       pub fn sanitize_description(server_name: &str, raw: &str) -> String {
           let cleaned = INSTRUCTION_PATTERN.replace_all(raw, "").to_string();
           // Collapse multiple whitespace
           let cleaned = cleaned.split_whitespace().collect::<Vec<_>>().join(" ");

           let prefix = format!("[External: {server_name}] ");
           let available = MAX_DESCRIPTION_LEN.saturating_sub(prefix.len());

           let body = if cleaned.len() > available {
               format!("{}...", &cleaned[..available.saturating_sub(3)])
           } else if cleaned.is_empty() {
               "No description provided".to_string()
           } else {
               cleaned
           };

           format!("{prefix}{body}")
       }

       /// Truncate a tool response if it exceeds the size cap.
       ///
       /// Returns the content unchanged if within the cap.
       /// If over the cap, truncates and appends "[truncated: N chars removed]".
       pub fn truncate_response(content: &str, cap: usize) -> String {
           if content.len() <= cap {
               return content.to_string();
           }
           let suffix = format!(" [truncated: {} chars removed]", content.len() - cap);
           let keep = cap.saturating_sub(suffix.len());
           format!("{}{}", &content[..keep], suffix)
       }
       ```
