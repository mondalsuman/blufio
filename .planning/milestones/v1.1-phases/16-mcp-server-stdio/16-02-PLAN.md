   ```rust
   pub struct BlufioMcpHandler {
       tool_registry: Arc<RwLock<ToolRegistry>>,
       export_tools: Vec<String>,
       timeout_secs: u64,
   }

   impl BlufioMcpHandler {
       pub fn new(
           tool_registry: Arc<RwLock<ToolRegistry>>,
           config: &McpConfig,
       ) -> Self {
           Self {
               tool_registry,
               export_tools: config.export_tools.clone(),
               timeout_secs: config.tool_timeout_secs,
           }
       }

       fn is_tool_exported(&self, name: &str) -> bool {
           // Use bridge::filtered_tool_names logic inline or delegate
           if name == "bash" { return false; }
           if self.export_tools.is_empty() { return true; }
           self.export_tools.contains(&name.to_string())
       }
   }
   ```

3. Implement `ServerHandler` for `BlufioMcpHandler`:

   - `initialize()`: Return `InitializeResult` with:
     - `capabilities.tools = Some(ToolsCapability::default())`
     - `capabilities.resources = None`, `capabilities.prompts = None`
     - `server_info = Implementation { name: "blufio", version: env!("CARGO_PKG_VERSION") }`
     - Call `context.peer.set_peer_info(request)` to store client info (following rmcp pattern)

   - `list_tools()`: Acquire read lock on tool_registry, use `bridge::filtered_tool_names` to get allowed names, then for each name get the tool and convert via `bridge::to_mcp_tool`. Return `ListToolsResult` with collected tools.

   - `call_tool()`:
     1. Check `is_tool_exported(name)` -> return `McpError::invalid_params(...)` if not
     2. Acquire read lock, get tool from registry -> return error if not found
     3. Get tool's parameters_schema, validate input via jsonschema:
        - If validation fails, collect error messages, return `McpError::invalid_params(format!("invalid input for '{}': {}", name, errors))`
        - The error code is -32602 (InvalidParams) per JSON-RPC spec
     4. Wrap `tool.invoke(input)` in `tokio::time::timeout(Duration::from_secs(self.timeout_secs), ...)`
     5. Match on timeout result:
        - `Ok(Ok(output))` -> `CallToolResult { content: vec![Content::text(output.content)], is_error: Some(output.is_error) }`
        - `Ok(Err(e))` -> `CallToolResult { content: vec![Content::text(format!("Tool error: {e}. Check input parameters."))], is_error: Some(true) }`
        - `Err(_elapsed)` -> `CallToolResult { content: vec![Content::text(format!("Tool '{}' timed out after {}s.", name, self.timeout_secs))], is_error: Some(true) }`
