4. Initialize minimal infrastructure (follow serve.rs pattern, stripped down):
   ```rust
   // Open database
   let db = open_db(&config).await?;

   // Vault startup check (for WASM skills that need secrets)
   let vault_conn = tokio_rusqlite::Connection::open(&config.storage.database_path).await
       .map_err(|e| BlufioError::Storage { source: Box::new(e) })?;
   match blufio_vault::vault_startup_check(vault_conn, &config.vault).await {
       Ok(Some(_vault)) => tracing::info!("vault unlocked"),
       Ok(None) => tracing::debug!("no vault found"),
       Err(e) => {
           tracing::error!(error = %e, "vault startup check failed");
           return Err(e);
       }
   }

   // Initialize tool registry with built-in tools
   let mut tool_registry = ToolRegistry::new();
   blufio_skill::builtin::register_builtins(&mut tool_registry);
   tracing::info!(count = tool_registry.len(), "tool registry initialized");
   let tool_registry = Arc::new(tokio::sync::RwLock::new(tool_registry));

   // Optionally load WASM skills if skill system enabled
   // (SkillProvider can register discovered skills into the registry)
   ```

5. Print startup banner to stderr:
   ```rust
   eprintln!("blufio {} MCP server ready", env!("CARGO_PKG_VERSION"));
   ```

6. Create handler and start stdio server:
   ```rust
   use rmcp::ServiceExt;

   let handler = BlufioMcpHandler::new(tool_registry, &config.mcp);
   let cancel = blufio_agent::shutdown::install_signal_handler();

   let service = handler
       .serve_with_ct((tokio::io::stdin(), tokio::io::stdout()), cancel.clone())
       .await
       .map_err(|e| BlufioError::Internal(format!("MCP server initialization failed: {e}")))?;

   // Wait for shutdown signal OR stdin EOF (client disconnected)
   tokio::select! {
       _ = cancel.cancelled() => {
           tracing::info!("MCP server shutting down (signal received)");
       }
       _ = service.waiting() => {
           tracing::info!("MCP server shutting down (client disconnected)");
       }
   }

   // Clean close
   db.close().await?;

   tracing::info!("MCP server shutdown complete");
   Ok(())
   ```

7. Important: The `open_db` function is defined in main.rs. Either:

   - Duplicate the simple 2-line function in mcp_server.rs

8. Ensure no `println!()` or `print!()` calls anywhere in this module -- only `eprintln!()` and `tracing::*` (which go to stderr via the configured subscriber).
