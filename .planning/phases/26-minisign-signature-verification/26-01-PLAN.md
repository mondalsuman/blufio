**Public API:**
```rust
use std::path::{Path, PathBuf};

/// Minisign public key for verifying Blufio releases.
/// Generated with: minisign -G -p blufio.pub -s blufio.key
const MINISIGN_PUBLIC_KEY: &str = "<base64 key from minisign -G>";

/// Result of a successful signature verification.
#[derive(Debug)]
pub struct VerifyResult {
    /// Name of the verified file.
    pub file_name: String,
    /// Trusted comment from the signature (cryptographically bound).
    pub trusted_comment: String,
}

/// Errors from signature verification.
#[derive(Debug)]
pub enum VerifyError {
    /// The file to verify was not found.
    FileNotFound { path: String },
    /// The signature file was not found.
    SignatureNotFound { path: String, hint: Option<String> },
    /// The signature file has invalid format.
    InvalidSignature { message: String },
    /// The signature does not match the file content.
    VerificationFailed { file_name: String, message: String },
    /// The embedded public key is invalid (should never happen).
    InvalidKey(String),
}
```

Implement `std::fmt::Display` for `VerifyError` with clear, actionable messages:
- `FileNotFound`: "File not found: {path}"
- `SignatureNotFound` (with hint): "Signature file not found: {path}\n  Use --signature <path> to specify manually"
- `SignatureNotFound` (no hint): "Signature file not found: {path}"
- `InvalidSignature`: "Invalid signature format: {message}"
- `VerificationFailed`: "Signature verification failed for '{file_name}': {message}"
- `InvalidKey`: "Invalid embedded public key: {message}"

Implement `std::error::Error` for `VerifyError`.
