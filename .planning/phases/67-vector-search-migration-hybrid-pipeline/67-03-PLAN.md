**Test 1: `test_vec0_parity_10_memories`**

Validates vec0 vs in-memory parity at 10-entry scale:

1. Create two stores sharing the same DB schema pattern: one with vec0_enabled=true, one with vec0_enabled=false.
2. Insert 10 memories with unique `synthetic_embedding(seed)` vectors, varied sources (Explicit, Extracted, FileWatcher), and varied created_at timestamps (0-30 days ago) for temporal decay differentiation.
3. For the vec0 store, call `populate_vec0()` to sync.
4. Run the same query through both stores using `vec0_search` (vec0 path) and `get_active_embeddings` + manual cosine (in-memory path).
5. Compare result ID sets: must be identical (same IDs in top-K).
6. Compare similarity scores: must be within 0.01 tolerance.
7. Note: Do NOT use HybridRetriever directly (requires OnnxEmbedder with ONNX model). Instead, test the scoring pipeline components: vec0_search vs in-memory cosine, then manually apply RRF fusion, importance boost, temporal decay to both result sets and compare.

```rust
#[tokio::test]
async fn test_vec0_parity_10_memories() {
    vec0::ensure_sqlite_vec_registered();
    let conn = setup_test_db().await;
    let store = Arc::new(MemoryStore::with_vec0(conn, None, true));

    // Insert 10 memories with varied embeddings, sources, and timestamps
    let now = chrono::Utc::now();
    let sources = ["explicit", "extracted", "filewatcher"];
    for i in 0..10 {
        let days_ago = (i * 3) as i64; // 0, 3, 6, ..., 27 days ago
        let created = (now - chrono::Duration::days(days_ago)).to_rfc3339();
        let emb = synthetic_embedding(i as u32);
        let source = sources[i % 3];
        let mem = make_test_memory(
            &format!("parity-{i}"),
            &format!("Memory content for parity test {i}"),
            emb,
            source,
            Some("test-session"),
        );
        // Override created_at
        let mut mem = mem;
        mem.created_at = created.clone();
        mem.updated_at = created;
        store.save(&mem).await.unwrap();
    }

    // Populate vec0 (should already be synced via dual-write, but ensure)
    let (populated, total) = store.populate_vec0().await.unwrap();
    assert_eq!(total, 10);

    // Query with a specific embedding
    let query_emb = synthetic_embedding(0);

    // Vec0 path
    let vec0_results = store.conn()
        .call({
            let q = query_emb.clone();
            move |conn| vec0::vec0_search(conn, &q, 10, 0.0, None)
        })
        .await.unwrap();

    // In-memory path
    let all_embeddings = store.get_active_embeddings().await.unwrap();
    let mut in_mem_results: Vec<(String, f32)> = all_embeddings.into_iter()
        .filter_map(|(id, emb)| {
            let sim = cosine_similarity(&query_emb, &emb);
            if sim >= 0.0 { Some((id, sim)) } else { None }
        })
        .collect();
    in_mem_results.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap_or(std::cmp::Ordering::Equal));

    // Compare: same IDs
    let vec0_ids: Vec<&str> = vec0_results.iter().map(|r| r.memory_id.as_str()).collect();
    let in_mem_ids: Vec<&str> = in_mem_results.iter().map(|(id, _)| id.as_str()).collect();
    assert_eq!(vec0_ids.len(), in_mem_ids.len(), "result count mismatch");

    // Same ID sets (order may vary for tied scores)
    let mut vec0_sorted = vec0_ids.clone();
    vec0_sorted.sort();
    let mut in_mem_sorted = in_mem_ids.clone();
    in_mem_sorted.sort();
    assert_eq!(vec0_sorted, in_mem_sorted, "result ID sets differ");

    // Scores within tolerance
    for (v, im) in vec0_results.iter().zip(in_mem_results.iter()) {
        assert!(
            (v.similarity - im.1).abs() < 0.01,
            "score mismatch for {}: vec0={}, in_mem={}", v.memory_id, v.similarity, im.1
        );
    }
}
```
