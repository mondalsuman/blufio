2. **Wire HookManager in serve.rs:**

   Add import: `use blufio_hooks::HookManager;`

   **Placement:** After EventBus creation (line ~259), after CronScheduler init block (~1464), but BEFORE the main loop starts. Following the CronScheduler non-fatal init pattern.

   ```rust
   // --- Hook system ---
   // Spawn after EventBus, uses CancellationToken for graceful shutdown.
   let hook_manager = if config.hooks.enabled {
       // Validate hook event names
       let warnings = blufio_hooks::manager::validate_hook_events(&config.hooks);
       for w in &warnings {
           warn!("{}", w);
       }

       let manager = HookManager::new(&config.hooks);
       let hook_rx = event_bus.subscribe_reliable(256).await;
       let hook_bus = event_bus.clone();
       let hook_cancel = cancel.child_token();

       // Execute pre_start hooks synchronously before main loop
       manager.execute_lifecycle_hooks("pre_start", &event_bus).await;

       // Clone manager for the run loop (manager needs to be used for direct lifecycle hooks too)
       // Since HookManager::run takes self, create a second manager for the run loop
       let run_manager = HookManager::new(&config.hooks);
       tokio::spawn(async move {
           run_manager.run(hook_rx, hook_bus, hook_cancel).await;
       });
       info!(
           hooks = config.hooks.definitions.iter().filter(|d| d.enabled).count(),
           "hook system started"
       );
       Some(manager)
   } else {
       debug!("hook system disabled by configuration");
       None
   };
   ```

   **NOTE on manager cloning:** HookManager::run takes `self` (not `&self`). We need the manager alive for direct lifecycle hooks (post_start, pre_shutdown, post_shutdown). Two approaches:
   - Option A: Create two HookManager instances from same config (one for run loop, one for direct calls)
   - Option B: Change HookManager::run to take `&self` with `Arc<Self>` pattern

   Use Option A (simpler, matches how CronScheduler is consumed). The run_manager handles EventBus-driven hooks, the hook_manager handles direct lifecycle hooks.

   **After all init completes, fire post_start:**
   ```rust
   // Fire post_start hooks after all subsystems are initialized
   if let Some(ref hm) = hook_manager {
       hm.execute_lifecycle_hooks("post_start", &event_bus).await;
   }
   ```

   async fn check_hooks(config: &BlufioConfig) -> CheckResult {
       if !config.hooks.enabled {
           return CheckResult {
               name: "Hook System".into(),
               status: CheckStatus::Warn,
               message: "Hook system is disabled".into(),
               details: Some("Enable with [hooks] enabled = true in blufio.toml".into()),
           };
       }

       let enabled_hooks: Vec<_> = config.hooks.definitions.iter()
           .filter(|d| d.enabled)
           .collect();

       if enabled_hooks.is_empty() {
           return CheckResult {
               name: "Hook System".into(),
               status: CheckStatus::Warn,
               message: "Hook system enabled but no hooks defined".into(),
               details: Some("Add [[hooks.definitions]] entries to blufio.toml".into()),
           };
       }

       // Validate hook event names
       let warnings = blufio_hooks::manager::validate_hook_events(&config.hooks);
       if !warnings.is_empty() {
           return CheckResult {
               name: "Hook System".into(),
               status: CheckStatus::Fail,
               message: format!("{} hook(s) reference unknown events", warnings.len()),
               details: Some(warnings.join("; ")),
           };
       }

       // Check hook commands exist
       let mut missing_commands = Vec::new();
       for hook in &enabled_hooks {
           // Extract first word of command as the executable
           let executable = hook.command.split_whitespace().next().unwrap_or(&hook.command);
           let exec_path = std::path::Path::new(executable);
           if exec_path.is_absolute() && !exec_path.exists() {
               missing_commands.push(format!("{}: {}", hook.name, executable));
           }
       }

       if !missing_commands.is_empty() {
           return CheckResult {
               name: "Hook System".into(),
               status: CheckStatus::Warn,
               message: format!("{} hook command(s) not found at absolute path", missing_commands.len()),
               details: Some(missing_commands.join("; ")),
           };
       }

       CheckResult {
           name: "Hook System".into(),
           status: CheckStatus::Pass,
           message: format!("{} hook(s) configured", enabled_hooks.len()),
           details: None,
       }
   }
   ```

2. **Add hot reload health check:**

   fn check_hot_reload(config: &BlufioConfig) -> CheckResult {
       if !config.hot_reload.enabled {
           return CheckResult {
               name: "Hot Reload".into(),
               status: CheckStatus::Warn,
               message: "Hot reload is disabled".into(),
               details: Some("Enable with [hot_reload] enabled = true in blufio.toml".into()),
           };
       }

       let mut features = Vec::new();
       features.push(format!("config reload ({}ms debounce)", config.hot_reload.debounce_ms));

       if config.hot_reload.tls_cert_path.is_some() && config.hot_reload.tls_key_path.is_some() {
           let cert_path = config.hot_reload.tls_cert_path.as_deref().unwrap_or("");
           let key_path = config.hot_reload.tls_key_path.as_deref().unwrap_or("");
           if std::path::Path::new(cert_path).exists() && std::path::Path::new(key_path).exists() {
               features.push("TLS cert reload".into());
           } else {
               return CheckResult {
                   name: "Hot Reload".into(),
                   status: CheckStatus::Fail,
                   message: "TLS cert/key paths configured but files not found".into(),
                   details: Some(format!("cert: {}, key: {}", cert_path, key_path)),
               };
           }
       }

       if config.hot_reload.watch_skills {
           let skills_dir = std::path::Path::new(&config.skill.skills_dir);
           if skills_dir.exists() {
               features.push("skill directory watching".into());
           } else {
               features.push(format!("skill watching configured but dir missing: {}", config.skill.skills_dir));
           }
       }

       CheckResult {
           name: "Hot Reload".into(),
           status: CheckStatus::Pass,
           message: format!("Enabled: {}", features.join(", ")),
           details: None,
       }
   }
   ```
