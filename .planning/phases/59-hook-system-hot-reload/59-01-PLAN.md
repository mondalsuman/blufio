1. **Create crates/blufio-hooks/src/recursion.rs:**
   RecursionGuard RAII pattern with AtomicU32:
   ```rust
   use std::sync::atomic::{AtomicU32, Ordering};
   use std::sync::Arc;

   pub struct RecursionGuard {
       counter: Arc<AtomicU32>,
   }

   impl RecursionGuard {
       /// Try to enter a recursion level. Returns None if max_depth exceeded.
       pub fn try_enter(counter: Arc<AtomicU32>, max_depth: u32) -> Option<Self> {
           let prev = counter.fetch_add(1, Ordering::SeqCst);
           if prev >= max_depth {
               counter.fetch_sub(1, Ordering::SeqCst);
               return None;
           }
           Some(Self { counter })
       }

       /// Current recursion depth.
       pub fn depth(counter: &AtomicU32) -> u32 {
           counter.load(Ordering::SeqCst)
       }
   }

   impl Drop for RecursionGuard {
       fn drop(&mut self) {
           self.counter.fetch_sub(1, Ordering::SeqCst);
       }
   }
   ```

   Tests:
   - `try_enter` succeeds at depth 0 with max_depth 3
   - `try_enter` succeeds up to max_depth - 1
   - `try_enter` returns None at max_depth
   - Drop decrements counter
   - Counter returns to 0 after all guards dropped
