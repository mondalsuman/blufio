   ```rust
   async fn check_encryption(db_path: &str) -> CheckResult {
       let start = Instant::now();
       let path = std::path::Path::new(db_path);

       if !path.exists() {
           return CheckResult {
               name: "Encryption".into(), status: CheckStatus::Pass,
               message: "no database yet".into(), duration: start.elapsed()
           };
       }

       let is_plaintext = blufio_storage::is_plaintext_sqlite(path).unwrap_or(true);
       let has_key = std::env::var("BLUFIO_DB_KEY").is_ok();

       match (is_plaintext, has_key) {
           (true, false) => CheckResult {
               name: "Encryption".into(), status: CheckStatus::Pass,
               message: "not encrypted".into(), duration: start.elapsed()
           },
           (true, true) => CheckResult {
               name: "Encryption".into(), status: CheckStatus::Warn,
               message: "BLUFIO_DB_KEY set but database is plaintext -- run: blufio db encrypt".into(),
               duration: start.elapsed()
           },
           (false, false) => CheckResult {
               name: "Encryption".into(), status: CheckStatus::Fail,
               message: "database is encrypted but BLUFIO_DB_KEY is not set".into(),
               duration: start.elapsed()
           },
           (false, true) => {
               // Query cipher details
               match blufio_storage::open_connection(db_path).await {
                   Ok(conn) => {
                       let info = conn.call(|conn| {
                           let version: String = conn
                               .query_row("PRAGMA cipher_version;", [], |row| row.get(0))
                               .unwrap_or_else(|_| "unknown".to_string());
                           let page_size: i64 = conn
                               .query_row("PRAGMA cipher_page_size;", [], |row| row.get(0))
                               .unwrap_or(4096);
                           Ok::<(String, i64), rusqlite::Error>((version, page_size))
                       }).await;

                       match info {
                           Ok((version, page_size)) => CheckResult {
                               name: "Encryption".into(), status: CheckStatus::Pass,
                               message: format!("encrypted (SQLCipher {version}, page size: {page_size})"),
                               duration: start.elapsed()
                           },
                           Err(_) => CheckResult {
                               name: "Encryption".into(), status: CheckStatus::Warn,
                               message: "encrypted but could not query cipher details".into(),
                               duration: start.elapsed()
                           },
                       }
                   },
                   Err(_) => CheckResult {
                       name: "Encryption".into(), status: CheckStatus::Fail,
                       message: "encrypted but cannot open -- verify BLUFIO_DB_KEY".into(),
                       duration: start.elapsed()
                   },
               }
           },
       }
   }
   ```
