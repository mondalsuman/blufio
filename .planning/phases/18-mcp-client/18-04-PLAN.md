    1. **doctor.rs**: Add a new check function for MCP servers:
       ```rust
       /// Check connectivity to configured MCP servers.
       #[cfg(feature = "mcp-client")]
       async fn check_mcp_servers(config: &BlufioConfig) -> Vec<CheckResult> {
           let mut results = Vec::new();

           if config.mcp.servers.is_empty() {
               return results;
           }

           for server in &config.mcp.servers {
               let start = Instant::now();
               let check_name = format!("mcp:{}", server.name);

               // Skip servers with invalid transport (would be caught by validation anyway)
               if !["http", "sse"].contains(&server.transport.as_str()) {
                   results.push(CheckResult {
                       name: check_name,
                       status: CheckStatus::Fail,
                       message: format!("invalid transport '{}'", server.transport),
                       duration: start.elapsed(),
                   });
                   continue;
               }

               // Attempt connection with timeout
               match tokio::time::timeout(
                   Duration::from_secs(server.connect_timeout_secs),
                   attempt_mcp_connection(server),
               ).await {
                   Ok(Ok(tool_count)) => {
                       results.push(CheckResult {
                           name: check_name,
                           status: CheckStatus::Pass,
                           message: format!(
                               "{} tools via {}",
                               tool_count, server.transport
                           ),
                           duration: start.elapsed(),
                       });
                   }
                   Ok(Err(e)) => {
                       results.push(CheckResult {
                           name: check_name,
                           status: CheckStatus::Fail,
                           message: format!("connection failed: {}", e),
                           duration: start.elapsed(),
                       });
                   }
                   Err(_) => {
                       results.push(CheckResult {
                           name: check_name,
                           status: CheckStatus::Fail,
                           message: format!(
                               "connection timed out after {}s",
                               server.connect_timeout_secs
                           ),
                           duration: start.elapsed(),
                       });
                   }
               }
           }

           results
       }

       #[cfg(feature = "mcp-client")]
       async fn attempt_mcp_connection(
           server: &blufio_config::model::McpServerEntry,
       ) -> Result<usize, String> {
           // Use the same connection logic as the client manager
           // but simplified for diagnostics -- just connect and list tools
           use rmcp::service::ServiceExt;

           let url = server.url.as_deref().ok_or("no URL configured")?;

           let service = match server.transport.as_str() {
               "http" => {
                   use rmcp::transport::StreamableHttpClientTransport;
                   use rmcp::transport::streamable_http_client::StreamableHttpClientTransportConfig;
                   let mut config = StreamableHttpClientTransportConfig::with_uri(url);
                   if let Some(ref token) = server.auth_token {
                       config = config.auth_header(token.as_str());
                   }
                   let transport = StreamableHttpClientTransport::new_with_config(config);
                   ().serve(transport).await.map_err(|e| format!("{e}"))?
               }
               "sse" => {
                   use rmcp::transport::SseClientTransport;
                   let transport = SseClientTransport::from_uri(url).await.map_err(|e| format!("{e}"))?;
                   ().serve(transport).await.map_err(|e| format!("{e}"))?
               }
               other => return Err(format!("unsupported transport: {other}")),
           };

           let tools = service.list_tools(Default::default()).await.map_err(|e| format!("{e}"))?;
           let count = tools.tools.len();

           // Clean disconnect
           let _ = service.cancel().await;

           Ok(count)
       }
       ```
