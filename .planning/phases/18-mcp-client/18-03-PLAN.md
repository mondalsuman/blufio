    2. **pin_store.rs**: Create SQLite-backed hash pin storage:
       ```rust
       use tokio_rusqlite::Connection;
       use tracing::{error, info, warn};

       /// SQLite-backed storage for MCP tool hash pins.
       ///
       /// Each tool definition is SHA-256 hashed at discovery time.
       /// On subsequent discoveries, the hash is compared to detect
       /// schema mutations (rug pulls).
       pub struct PinStore {
           conn: Connection,
       }

       impl PinStore {
           /// Open or create the pin store, creating the table if needed.
           pub async fn open(db_path: &str) -> Result<Self, Box<dyn std::error::Error + Send + Sync>> {
               let conn = Connection::open(db_path).await?;
               conn.call(|conn| {
                   conn.execute_batch(
                       "CREATE TABLE IF NOT EXISTS mcp_tool_pins (
                           server_name TEXT NOT NULL,
                           tool_name TEXT NOT NULL,
                           pin_hash TEXT NOT NULL,
                           created_at TEXT NOT NULL DEFAULT (datetime('now')),
                           updated_at TEXT NOT NULL DEFAULT (datetime('now')),
                           PRIMARY KEY (server_name, tool_name)
                       )"
                   )?;
                   Ok(())
               }).await?;
               Ok(Self { conn })
           }

           /// Store a hash pin for a tool.
           pub async fn store_pin(&self, server: &str, tool: &str, hash: &str) -> Result<(), Box<dyn std::error::Error + Send + Sync>> { ... }

           /// Get the stored pin for a tool, if any.
           pub async fn get_pin(&self, server: &str, tool: &str) -> Result<Option<String>, Box<dyn std::error::Error + Send + Sync>> { ... }

           /// Delete a pin (used by `blufio mcp re-pin` CLI).
           pub async fn delete_pin(&self, server: &str, tool: &str) -> Result<bool, Box<dyn std::error::Error + Send + Sync>> { ... }

           /// List all pins for a server (for diagnostics).
           pub async fn list_pins(&self, server: &str) -> Result<Vec<(String, String)>, Box<dyn std::error::Error + Send + Sync>> { ... }
       }
       ```

       Use `conn.call(move |conn| { ... })` pattern for all SQLite operations (same as CostLedger pattern).

  <behavior>
    - Health monitor runs periodic ping checks for each connected server
    - Ping interval is configurable (default: 60 seconds per CONTEXT discretion)
    - Failed pings mark server as degraded and remove its tools from LLM context
    - Successful pings on degraded servers trigger auto-recovery (re-discover + re-register)
    - Reconnection uses exponential backoff: 1s, 2s, 4s, 8s, 16s, 32s, 60s cap
    - Trust zone section is added to agent context when external tools are registered
    - serve.rs spawns health monitor background task
  </behavior>
  <action>
    1. **health.rs**: Create the health check background task:
       ```rust
       use std::collections::HashMap;
       use std::sync::Arc;
       use std::time::Duration;

       use blufio_config::model::McpServerEntry;
       use blufio_skill::tool::ToolRegistry;
       use tokio::sync::RwLock;
       use tokio_util::sync::CancellationToken;
       use tracing::{info, warn, debug};

       const DEFAULT_HEALTH_INTERVAL_SECS: u64 = 60;
       const BACKOFF_BASE_SECS: u64 = 1;
       const BACKOFF_CAP_SECS: u64 = 60;

       /// Spawn a background task that periodically health-checks MCP servers.
       ///
       /// For each server:
       /// - If connected: ping to verify liveness
       /// - If degraded: attempt reconnection with exponential backoff
       /// - On recovery: re-discover tools and re-register in ToolRegistry
       pub fn spawn_health_monitor(
           servers: Vec<McpServerEntry>,
           tool_registry: Arc<RwLock<ToolRegistry>>,
           cancel: CancellationToken,
       ) -> tokio::task::JoinHandle<()> {
           tokio::spawn(async move {
               let mut interval = tokio::time::interval(
                   Duration::from_secs(DEFAULT_HEALTH_INTERVAL_SECS)
               );
               // Track backoff per server
               let mut backoff_secs: HashMap<String, u64> = HashMap::new();

               // Skip initial tick
               interval.tick().await;

               loop {
                   tokio::select! {
                       _ = interval.tick() => {
                           // Health check each server
                           for server in &servers {
                               let current_backoff = backoff_secs
                                   .get(&server.name)
                                   .copied()
                                   .unwrap_or(0);

                               if current_backoff > 0 {
                                   // Server is in backoff -- decrement and skip
                                   let new_backoff = current_backoff.saturating_sub(
                                       DEFAULT_HEALTH_INTERVAL_SECS
                                   );
                                   backoff_secs.insert(server.name.clone(), new_backoff);
                                   if new_backoff > 0 {
                                       debug!(
                                           server = %server.name,
                                           remaining_backoff_secs = new_backoff,
                                           "MCP server in backoff, skipping health check"
                                       );
                                       continue;
                                   }
                               }

                               // Attempt ping/reconnect logic here
                               // On failure: set backoff
                               // On success: clear backoff
                               // This is a simplified version -- the actual implementation
                               // should track ServerState and attempt reconnection for
                               // degraded servers.
                               debug!(server = %server.name, "MCP health check");
                           }
                       }
                       _ = cancel.cancelled() => {
                           info!("MCP health monitor shutting down");
                           break;
                       }
                   }
               }
           })
       }
       ```
