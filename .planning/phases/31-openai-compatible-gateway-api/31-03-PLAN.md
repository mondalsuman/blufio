```rust
use serde::{Deserialize, Serialize};

/// Query parameters for GET /v1/tools.
#[derive(Debug, Deserialize)]
pub struct ToolsQueryParams {
    /// Filter by tool source: "builtin", "wasm", "mcp".
    #[serde(default)]
    pub source: Option<String>,
}

/// Response for GET /v1/tools.
#[derive(Debug, Serialize)]
pub struct ToolListResponse {
    pub object: String,  // "list"
    pub data: Vec<ToolInfo>,
}

/// Extended tool info in OpenAI function schema format.
#[derive(Debug, Serialize)]
pub struct ToolInfo {
    /// Tool type (always "function").
    #[serde(rename = "type")]
    pub tool_type: String,
    /// Function definition.
    pub function: ToolFunctionInfo,
    /// Extended: tool source.
    pub source: String,  // "builtin", "wasm", "mcp"
    /// Extended: tool version (if available).
    #[serde(skip_serializing_if = "Option::is_none")]
    pub version: Option<String>,
    /// Extended: required permissions/capabilities.
    #[serde(skip_serializing_if = "Option::is_none")]
    pub required_permissions: Option<Vec<String>>,
}

/// Function info within a ToolInfo.
#[derive(Debug, Serialize)]
pub struct ToolFunctionInfo {
    pub name: String,
    pub description: String,
    pub parameters: serde_json::Value,
}

/// Request body for POST /v1/tools/invoke.
#[derive(Debug, Deserialize)]
pub struct ToolInvokeRequest {
    /// Tool name to invoke.
    pub name: String,
    /// JSON input to pass to the tool.
    #[serde(default = "default_empty_object")]
    pub input: serde_json::Value,
}

fn default_empty_object() -> serde_json::Value {
    serde_json::json!({})
}

/// Response body for POST /v1/tools/invoke.
#[derive(Debug, Serialize)]
pub struct ToolInvokeResponse {
    /// Tool name that was invoked.
    pub name: String,
    /// Tool execution output.
    pub output: String,
    /// Whether the execution resulted in an error.
    pub is_error: bool,
    /// Execution time in milliseconds.
    #[serde(skip_serializing_if = "Option::is_none")]
    pub duration_ms: Option<u64>,
}
```

Add `pub mod tools_types;` to `openai_compat/mod.rs`.

**2. Implement handlers:**

Create `crates/blufio-gateway/src/openai_compat/tools.rs`:

```rust
use crate::openai_compat::tools_types::*;
use crate::openai_compat::types::GatewayErrorResponse;
use crate::server::GatewayState;

/// GET /v1/tools
pub async fn get_tools(
    State(state): State<GatewayState>,
    Query(params): Query<ToolsQueryParams>,
) -> Response
```

Logic:
1. If tools registry not set, return empty list
2. Read tool registry: `let registry = state.tools.as_ref().unwrap().read().await;`
3. Get tool definitions: `registry.tool_definitions()`
4. Filter by allowlist: only include tools whose name is in `state.api_tools_allowlist`
5. Filter by source param if present
6. Map each `ToolDefinition` to `ToolInfo`:
   - `type`: "function"
   - `function`: { name, description, parameters: input_schema }
   - `source`: Determine from tool name — built-in tools (bash, http, file) are "builtin", namespaced tools (prefix__name) check namespace: "mcp" or "wasm"
   - `version`: None for builtins, could be extracted from WASM manifests (optional)
7. Return `ToolListResponse { object: "list", data: tool_infos }`

```rust
/// POST /v1/tools/invoke
pub async fn post_tool_invoke(
    State(state): State<GatewayState>,
    Json(body): Json<ToolInvokeRequest>,
) -> Response
```

Logic:
1. Check tools registry exists, return 503 if not
2. Check tool name is in allowlist, return 403 if not:
   ```json
   {"error": {"message": "Tool 'xxx' is not in the API tools allowlist", "type": "permission_denied", "code": "tool_not_allowed"}}
   ```
3. Get tool from registry: `registry.get(&body.name)`
4. If None, return 404:
   ```json
   {"error": {"message": "Tool 'xxx' not found", "type": "not_found", "code": "tool_not_found"}}
   ```
5. Execute tool: `let start = Instant::now(); let result = tool.execute(body.input).await;`
6. Map result to ToolInvokeResponse:
   - On Ok(output): `{ name, output: output.content, is_error: output.is_error, duration_ms }`
   - On Err(e): `{ name, output: e.to_string(), is_error: true, duration_ms }`
7. Return JSON response
