2. In external_tool.rs, add W3C traceparent injection on outbound MCP tool calls.
   Read the file first to find the HTTP request construction point.

   Add a cfg-gated function:
   ```rust
   #[cfg(feature = "otel")]
   fn inject_trace_context(headers: &mut reqwest::header::HeaderMap) {
       use opentelemetry::propagation::Injector;

       struct HeaderMapInjector<'a> {
           headers: &'a mut reqwest::header::HeaderMap,
       }

       impl<'a> Injector for HeaderMapInjector<'a> {
           fn set(&mut self, key: &str, value: String) {
               if let Ok(name) = reqwest::header::HeaderName::from_bytes(key.as_bytes()) {
                   if let Ok(val) = reqwest::header::HeaderValue::from_str(&value) {
                       self.headers.insert(name, val);
                   }
               }
           }
       }

       opentelemetry::global::get_text_map_propagator(|propagator| {
           propagator.inject_context(
               &opentelemetry::Context::current(),
               &mut HeaderMapInjector { headers },
           );
       });
   }
   ```

   Call inject_trace_context(&mut headers) before sending HTTP requests to MCP servers.
   Wrap in span: tracing::info_span!("blufio.mcp.call", "tool_name" = %tool_name, "server_name" = %server_name)

4. In blufio-gateway/src/server.rs, add X-Trace-Id response header middleware:
   ```rust
   // Add as a layer on the app router (after all routes merged)
   async fn trace_id_header_middleware(
       request: axum::http::Request<axum::body::Body>,
       next: axum::middleware::Next,
   ) -> axum::response::Response {
       let mut response = next.run(request).await;

       // Extract current trace ID from tracing span context
       #[cfg(feature = "otel")]
       {
           use tracing_opentelemetry::OpenTelemetrySpanExt;
           let context = tracing::Span::current().context();
           let span_ref = context.span();
           let span_context = span_ref.span_context();
           if span_context.is_valid() {
               if let Ok(val) = axum::http::HeaderValue::from_str(&span_context.trace_id().to_string()) {
                   response.headers_mut().insert("x-trace-id", val);
               }
           }
       }

       response
   }
   ```
